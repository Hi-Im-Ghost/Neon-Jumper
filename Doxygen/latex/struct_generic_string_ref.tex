\doxysubsection{Detailed Description}
\subsubsection*{template$<$typename Char\+Type$>$\newline
struct Generic\+String\+Ref$<$ Char\+Type $>$}
Reference to a constant string (not taking a copy) 


\begin{DoxyTemplParams}{Template Parameters}
{\em Char\+Type} & character type of the string\\
\hline
\end{DoxyTemplParams}
This helper class is used to automatically infer constant string references for string literals, especially from {\ttfamily const} {\bfseries{}}(!) character arrays.

The main use is for creating JSON string values without copying the source string via an \mbox{\hyperlink{md_src_rapidjson_doc_dom_zh_cn_Allocator}{Allocator}}. This requires that the referenced string pointers have a sufficient lifetime, which exceeds the lifetime of the associated \mbox{\hyperlink{class_generic_value}{Generic\+Value}}.

{\bfseries{Example}} 
\begin{DoxyCode}{0}
\DoxyCodeLine{\mbox{\hyperlink{class_generic_value}{Value}} v(\textcolor{stringliteral}{"{}foo"{}});   \textcolor{comment}{// ok, no need to copy \& calculate length}}
\DoxyCodeLine{\textcolor{keyword}{const} \textcolor{keywordtype}{char} foo[] = \textcolor{stringliteral}{"{}foo"{}};}
\DoxyCodeLine{v.SetString(foo); \textcolor{comment}{// ok}}
\DoxyCodeLine{}
\DoxyCodeLine{\textcolor{keyword}{const} \textcolor{keywordtype}{char}* bar = foo;}
\DoxyCodeLine{\textcolor{comment}{// Value x(bar); // not ok, can't rely on bar's lifetime}}
\DoxyCodeLine{\mbox{\hyperlink{class_generic_value}{Value}} x(\mbox{\hyperlink{struct_generic_string_ref_aa6b9fd9f6aa49405a574c362ba9af6b5}{StringRef}}(bar)); \textcolor{comment}{// lifetime explicitly guaranteed by user}}
\DoxyCodeLine{\mbox{\hyperlink{class_generic_value}{Value}} y(\mbox{\hyperlink{struct_generic_string_ref_aa6b9fd9f6aa49405a574c362ba9af6b5}{StringRef}}(bar, 3));  \textcolor{comment}{// ok, explicitly pass length}}

\end{DoxyCode}
